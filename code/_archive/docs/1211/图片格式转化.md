方法 2：命令行（批量处理快）

  # macOS 安装工具
  brew install webp

  # 转换单个文件
  cwebp -q 90 home-bg.png -o home-bg.webp

  # 批量转换所有 PNG
  for f in *.png; do
    cwebp -q 90 "$f" -o "${f%.png}.webp"
  done

  10张图 10 秒搞定！
